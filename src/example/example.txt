// CIE Pseudocode Calculator Program

// Function to perform the calculation
CONSTANT date <- 01/01/2025

FUNCTION Calculate(num1 : REAL, op : CHAR, num2 : REAL) RETURNS REAL
    DECLARE return_value : REAL
    CASE OF op
        '+':
            return_value <- num1 + num2
        '-':
            return_value <- num1 - num2
        '*':
            return_value <- num1 * num2
        '/':
            IF num2 = 0 THEN
                OUTPUT "Error: Division by zero."
                return_value <- 0
            ELSE
                return_value <- num1 / num2
            ENDIF
        OTHERWISE:
            OUTPUT "Error: Invalid operator."
            return_value <- 0
    ENDCASE
    RETURN return_value
ENDFUNCTION

// Procedure to display the result
PROCEDURE DisplayResult(num1 : REAL, op : CHAR, num2 : REAL, result : REAL)
    OUTPUT num1, " ", op, " ", num2, " = ", result
ENDPROCEDURE

TYPE Log
    DECLARE date : DATE
    DECLARE result : REAL
ENDTYPE

// Main program
DECLARE num1, num2, result : REAL
DECLARE op : CHAR
DECLARE continueCalc : STRING
DECLARE history : ARRAY[0:100] OF Log
DECLARE matrix : ARRAY[0:10,0:10] OF REAL
DECLARE index : INTEGER
index <- 0

REPEAT
    OUTPUT "Enter the first number:"
    INPUT num1
    OUTPUT "Enter an operator (+, -, *, /):"
    INPUT op
    OUTPUT "Enter the second number:"
    INPUT num2

    result <- Calculate(num1, op, num2)
    IF index = 101 THEN
        FOR i <- 10 TO 100
            history[i - 10] <- history[i]
            FOR j <- 0 TO 3
                OUTPUT "..."
            NEXT j
        NEXT i
        index <- 91
    ENDIF
    DECLARE new_log : Log
    new_log.date <- 01/01/2026
    new_log.result <- result
    history[index] <- new_log
    CALL DisplayResult(num1, op, num2, result)

    OUTPUT "Do you want to perform another calculation? (yes/no):"
    INPUT continueCalc
UNTIL NOT LCASE(continueCalc) = "yes"

// End of program
