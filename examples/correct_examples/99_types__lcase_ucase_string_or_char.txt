// LCASE/UCASE project decision demo
// Accepts CHAR or STRING and returns the same type.
// BEHAVIOR_CAPTURED_BY_OUTPUT: LCASE(STRING) prints lowercase STRING; UCASE(CHAR) prints uppercase CHAR ("hello" and A)

DECLARE s : STRING
DECLARE c : CHAR

s <- "Hello"
c <- 'a'

DECLARE s2 : STRING
DECLARE c2 : CHAR

s2 <- LCASE(s)
c2 <- UCASE(c)

OUTPUT s2
OUTPUT c2
