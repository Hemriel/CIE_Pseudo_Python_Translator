// Working example of for loops

DECLARE i : INTEGER
DECLARE j : INTEGER
DECLARE start : INTEGER
DECLARE finish : INTEGER
DECLARE total : INTEGER
DECLARE nums : ARRAY[1:5] OF INTEGER

// Example 1: Basic count-controlled loop (inclusive)
OUTPUT "Example 1: i from 1 to 5 (inclusive)"
FOR i <- 1 TO 5
    OUTPUT "i = ", i
NEXT i

// Example 2: Bounds are expressions that evaluate to integers
OUTPUT "Example 2: i from  2 + 1 to (10 DIV 2) + 3"
FOR i <- 2 + 1 TO (10 DIV 2) + 3
    OUTPUT "i = ", i
NEXT i

// Example 3: value1 = value2 -> executes exactly once
OUTPUT "Example 3: runs once (value1 = value2)"
FOR i <- 3 TO 3
    OUTPUT "This executes once; i = ", i
NEXT i

// Example 4: value1 > value2 -> executes zero times
OUTPUT "Example 4: does not execute (value1 > value2)"
FOR i <- 5 TO 3
    OUTPUT "You should not see this output."
NEXT i

// Example 5: Nested FOR loops (repeat the identifier after NEXT)
OUTPUT "Example 5: nested loops"
FOR i <- 1 TO 3
    OUTPUT "Outer i = ", i
    FOR j <- 1 TO 2
        OUTPUT "  Inner j = ", j
    NEXT j
NEXT i

// Example 6: Fill an array and compute a total
OUTPUT "Example 6: fill nums[1..5] with squares and sum them"
FOR i <- 1 TO 5
    nums[i] <- i * i
NEXT i

total <- 0
FOR i <- 1 TO 5
    total <- total + nums[i]
NEXT i
OUTPUT "Sum of squares = ", total
