// TEST: WHILE loops (normal + 0-iteration)
// EXPECTED: valid

DECLARE counter : INTEGER
counter <- 0

// Counts from 0 to 4 (condition checked before each iteration)
WHILE counter < 5
    OUTPUT "counter = ", counter
    counter <- counter + 1
ENDWHILE

// Demonstrates that the body will not run if the condition is FALSE initially
DECLARE x : INTEGER
x <- 10
WHILE x < 0
    OUTPUT "This will not be printed"
ENDWHILE
