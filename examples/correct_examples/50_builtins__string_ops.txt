// TEST: string/char built-ins + concatenation
// EXPECTED: valid
// BEHAVIOR_CAPTURED_BY_OUTPUT: string built-ins (RIGHT/LENGTH/MID), char built-ins (LCASE/UCASE), and concatenation results

DECLARE ThisString : STRING
DECLARE ResultString : STRING
DECLARE ResultLength : INTEGER
DECLARE ThisChar : CHAR

// RIGHT(ThisString : STRING, x : INTEGER) RETURNS STRING
ThisString <- "ABCDEFGH"
ResultString <- RIGHT(ThisString, 3)
OUTPUT "RIGHT(ABCDEFGH, 3) returns ", ResultString

// LENGTH(ThisString : STRING) RETURNS INTEGER
ResultLength <- LENGTH("Happy Days")
OUTPUT "LENGTH(Happy Days) returns ", ResultLength

// MID(ThisString : STRING, x : INTEGER, y : INTEGER) RETURNS STRING
ResultString <- MID("ABCDEFGH", 2, 3)
OUTPUT "MID(ABCDEFGH, 2, 3) returns ", ResultString

// LCASE(ThisChar : CHAR) RETURNS CHAR
ThisChar <- LCASE('W')
OUTPUT "LCASE(W) returns ", ThisChar

// UCASE(ThisChar : CHAR) RETURNS CHAR
ThisChar <- UCASE('h')
OUTPUT "UCASE(h) returns ", ThisChar

// Concatenation
OUTPUT "Hello + World = ", "Hello " & "World"
