DECLARE x : INTEGER

FUNCTION AddOne(n : INTEGER) RETURNS INTEGER
    RETURN n + 1
ENDFUNCTION

PROCEDURE Show(n : INTEGER)
    OUTPUT "Show: ", n
ENDPROCEDURE

x <- AddOne(4)
OUTPUT "x = ", x

CALL Show(x)
