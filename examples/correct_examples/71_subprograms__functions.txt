DECLARE a, b : INTEGER
DECLARE result : INTEGER

a <- 5
b <- 12

FUNCTION Five() RETURNS INTEGER
    RETURN 5
ENDFUNCTION

FUNCTION Add(x : INTEGER, y : INTEGER) RETURNS INTEGER
    RETURN x + y
ENDFUNCTION

FUNCTION Square(n : INTEGER) RETURNS INTEGER
    RETURN n * n
ENDFUNCTION

FUNCTION Max(x : INTEGER, y : INTEGER) RETURNS INTEGER
    IF x > y THEN
        RETURN x
    ELSE
        RETURN y
    ENDIF
ENDFUNCTION

PROCEDURE ShowNumber(n : INTEGER)
    OUTPUT "ShowNumber: ", n
ENDPROCEDURE

OUTPUT "a = ", a, ", b = ", b
OUTPUT "Add(a, b) = ", Add(a, b)
OUTPUT "Square(a) = ", Square(a)
OUTPUT "Five() = ", Five()
OUTPUT "Max(a, b) = ", Max(a, b)
